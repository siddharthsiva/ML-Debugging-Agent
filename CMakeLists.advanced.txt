# Advanced CMake configuration with quantum computing support
cmake_minimum_required(VERSION 3.20)
project(MiniSnowflake VERSION 2.0.0 LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_STANDARD_REQUIRED ON)

# Advanced optimization options
option(ENABLE_QUANTUM "Enable quantum computing features" ON)
option(ENABLE_NEUROMORPHIC "Enable neuromorphic computing" ON)
option(ENABLE_DNA_STORAGE "Enable DNA-inspired algorithms" ON)
option(ENABLE_HOLOGRAPHIC "Enable holographic storage" ON)
option(ENABLE_METAMATERIAL "Enable metamaterial data structures" ON)
option(ENABLE_CHAOS_THEORY "Enable chaos theory predictions" ON)
option(ENABLE_SWARM_INTELLIGENCE "Enable swarm optimization" ON)
option(ENABLE_TOPOLOGICAL "Enable topological computing" ON)
option(USE_INTEL_MKL "Use Intel Math Kernel Library" OFF)
option(USE_CUDA_QUANTUM "Use NVIDIA cuQuantum" OFF)
option(OPTIMIZE_FOR_NATIVE "Optimize for native architecture" ON)

# Find advanced packages
find_package(Threads REQUIRED)
find_package(PkgConfig REQUIRED)
find_package(CUDA TOOLKIT REQUIRED)

# Quantum computing libraries
if(ENABLE_QUANTUM)
    find_package(Qiskit QUIET)
    find_package(Cirq QUIET)
    find_path(CUQUANTUM_INCLUDE_DIR cuquantum.h)
    find_library(CUQUANTUM_LIBRARY cuquantum)
endif()

# SIMD and vectorization
if(OPTIMIZE_FOR_NATIVE)
    include(CheckCXXCompilerFlag)
    check_cxx_compiler_flag("-march=native" COMPILER_SUPPORTS_MARCH_NATIVE)
    if(COMPILER_SUPPORTS_MARCH_NATIVE)
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -march=native -mtune=native")
    endif()
    
    # Advanced SIMD support
    check_cxx_compiler_flag("-mavx512f" COMPILER_SUPPORTS_AVX512)
    if(COMPILER_SUPPORTS_AVX512)
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -mavx512f -mavx512cd")
    endif()
endif()

# Intel MKL for advanced mathematics
if(USE_INTEL_MKL)
    find_package(MKL REQUIRED)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DUSE_INTEL_MKL")
endif()

# Performance optimization flags
if(CMAKE_BUILD_TYPE MATCHES Release)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -flto -ffast-math -funroll-loops")
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -DEIGEN_NO_DEBUG -DNDEBUG")
    
    # Profile-guided optimization
    option(USE_PGO "Use Profile-Guided Optimization" OFF)
    if(USE_PGO)
        set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -fprofile-generate")
        set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -fprofile-generate")
    endif()
    
else()
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -g -O0 -fsanitize=address,undefined")
    set(CMAKE_LINKER_FLAGS "${CMAKE_LINKER_FLAGS} -fsanitize=address,undefined")
endif()

# Advanced compiler warnings
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall -Wextra -Wpedantic -Wconversion")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wshadow -Wnon-virtual-dtor -Wold-style-cast")

# Include directories
include_directories(${PROJECT_SOURCE_DIR}/include)
include_directories(${PROJECT_SOURCE_DIR}/external/eigen)
include_directories(${PROJECT_SOURCE_DIR}/external/boost)

if(CUQUANTUM_INCLUDE_DIR)
    include_directories(${CUQUANTUM_INCLUDE_DIR})
endif()

# Source file organization
file(GLOB_RECURSE CORE_SOURCES
    "src/types.cpp"
    "src/storage.cpp"
    "src/catalog.cpp"
    "src/parser.cpp"
    "src/query_engine.cpp"
    "src/server.cpp"
)

file(GLOB_RECURSE ADVANCED_SOURCES
    "src/advanced_features.cpp"
    "src/enterprise_features.cpp"
    "src/security_features.cpp"
)

file(GLOB_RECURSE QUANTUM_SOURCES
    "src/quantum_features.cpp"
    "src/neuromorphic_processor.cpp"
    "src/bio_inspired_storage.cpp"
    "src/swarm_optimizer.cpp"
    "src/chaos_predictor.cpp"
    "src/holographic_storage.cpp"
    "src/topological_processor.cpp"
)

set(ALL_SOURCES ${CORE_SOURCES} ${ADVANCED_SOURCES})

# Conditional compilation based on features
if(ENABLE_QUANTUM)
    list(APPEND ALL_SOURCES ${QUANTUM_SOURCES})
    add_compile_definitions(ENABLE_QUANTUM_FEATURES)
endif()

if(ENABLE_NEUROMORPHIC)
    add_compile_definitions(ENABLE_NEUROMORPHIC_FEATURES)
endif()

if(ENABLE_DNA_STORAGE)
    add_compile_definitions(ENABLE_DNA_STORAGE)
endif()

if(ENABLE_HOLOGRAPHIC)
    add_compile_definitions(ENABLE_HOLOGRAPHIC_STORAGE)
endif()

# Create main executable
add_executable(mini_snowflake src/main.cpp ${ALL_SOURCES})

# CUDA support for GPU acceleration
if(CMAKE_CUDA_COMPILER)
    set_property(TARGET mini_snowflake PROPERTY CUDA_ARCHITECTURES 70 75 80 86 90)
    target_link_libraries(mini_snowflake CUDA::cudart CUDA::cublas CUDA::cufft)
    
    if(CUQUANTUM_LIBRARY)
        target_link_libraries(mini_snowflake ${CUQUANTUM_LIBRARY})
    endif()
endif()

# Link libraries
target_link_libraries(mini_snowflake Threads::Threads)

if(USE_INTEL_MKL)
    target_link_libraries(mini_snowflake MKL::MKL)
endif()

# Advanced features testing
enable_testing()

add_executable(quantum_tests test/quantum_test.cpp ${QUANTUM_SOURCES})
target_link_libraries(quantum_tests Threads::Threads)
add_test(NAME QuantumFeatures COMMAND quantum_tests)

add_executable(neuromorphic_tests test/neuromorphic_test.cpp)
target_link_libraries(neuromorphic_tests Threads::Threads)
add_test(NAME NeuromorphicProcessing COMMAND neuromorphic_tests)

# Performance benchmarks
add_executable(performance_benchmark benchmark/performance_suite.cpp ${ALL_SOURCES})
target_link_libraries(performance_benchmark Threads::Threads)

# Documentation generation
find_package(Doxygen)
if(DOXYGEN_FOUND)
    configure_file(${CMAKE_CURRENT_SOURCE_DIR}/docs/Doxyfile.in 
                   ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
    add_custom_target(docs
        ${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
        WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
        COMMENT "Generating API documentation with Doxygen" VERBATIM)
endif()

# Installation
install(TARGETS mini_snowflake DESTINATION bin)
install(DIRECTORY include/ DESTINATION include/snowflake)
install(FILES README.md LICENSE DESTINATION share/doc/mini-snowflake)

# Packaging
set(CPACK_PACKAGE_NAME "Mini-Snowflake")
set(CPACK_PACKAGE_VENDOR "Quantum Analytics Inc.")
set(CPACK_PACKAGE_VERSION ${PROJECT_VERSION})
set(CPACK_PACKAGE_DESCRIPTION_SUMMARY "Quantum-Enhanced Analytical Database")
set(CPACK_PACKAGE_DESCRIPTION_FILE "${CMAKE_CURRENT_SOURCE_DIR}/README.md")
set(CPACK_RESOURCE_FILE_LICENSE "${CMAKE_CURRENT_SOURCE_DIR}/LICENSE")

include(CPack)
